213  // Helper function to initialise the container of buffers.
214  template <typename Iterator>
215  void init_buffers(execution_context& ctx, Iterator begin, Iterator end)
216  {
217  std::size_t n = std::distance(begin, end);
218  buffers_.resize(n);
219 
220 #if defined(ASIO_HAS_IO_URING)
221  service_ = &use_service<detail::io_uring_service>(ctx);
222  std::vector<iovec,
223  ASIO_REBIND_ALLOC(allocator_type, iovec)> iovecs(n,
224  ASIO_REBIND_ALLOC(allocator_type, iovec)(
225  buffers_.get_allocator()));
226 #endif // defined(ASIO_HAS_IO_URING)
227 
228  Iterator iter = begin;
229  for (int index = 0; iter != end; ++index, ++iter)
230  {
231  mutable_buffer b(*iter);
232  std::size_t i = static_cast<std::size_t>(index);
233  buffers_[i] = this->make_buffer(b, &ctx, index);
234 
235 #if defined(ASIO_HAS_IO_URING)
236  iovecs[i].iov_base = buffers_[i].data();
237  iovecs[i].iov_len = buffers_[i].size();
238 #endif // defined(ASIO_HAS_IO_URING)
239  }
240 
241 #if defined(ASIO_HAS_IO_URING)
242  if (n > 0)
243  {
244  service_->register_buffers(&iovecs[0],
245  static_cast<unsigned>(iovecs.size()));
246  }
247 #endif // defined(ASIO_HAS_IO_URING)
248  }
