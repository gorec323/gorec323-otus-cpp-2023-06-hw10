30 // Class to adapt an append_t as a completion handler.
31 template <typename Handler, typename... Values>
32 class append_handler
33 {
34 public:
35  typedef void result_type;
36 
37  template <typename H>
38  append_handler(H&& handler, std::tuple<Values...> values)
39  : handler_(static_cast<H&&>(handler)),
40  values_(static_cast<std::tuple<Values...>&&>(values))
41  {
42  }
43 
44  template <typename... Args>
45  void operator()(Args&&... args)
46  {
47  this->invoke(
48  index_sequence_for<Values...>{},
49  static_cast<Args&&>(args)...);
50  }
51 
52  template <std::size_t... I, typename... Args>
53  void invoke(index_sequence<I...>, Args&&... args)
54  {
55  static_cast<Handler&&>(handler_)(
56  static_cast<Args&&>(args)...,
57  static_cast<Values&&>(std::get<I>(values_))...);
58  }
59 
60 //private:
61  Handler handler_;
62  std::tuple<Values...> values_;
63 };
