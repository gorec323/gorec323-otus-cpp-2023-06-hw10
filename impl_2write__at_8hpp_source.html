38  template <typename SyncRandomAccessWriteDevice, typename ConstBufferSequence,
39  typename ConstBufferIterator, typename CompletionCondition>
40  std::size_t write_at_buffer_sequence(SyncRandomAccessWriteDevice& d,
41  uint64_t offset, const ConstBufferSequence& buffers,
42  const ConstBufferIterator&, CompletionCondition completion_condition,
43  asio::error_code& ec)
44  {
45  ec = asio::error_code();
46  asio::detail::consuming_buffers<const_buffer,
47  ConstBufferSequence, ConstBufferIterator> tmp(buffers);
48  while (!tmp.empty())
49  {
50  if (std::size_t max_size = detail::adapt_completion_condition_result(
51  completion_condition(ec, tmp.total_consumed())))
52  {
53  tmp.consume(d.write_some_at(offset + tmp.total_consumed(),
54  tmp.prepare(max_size), ec));
55  }
56  else
57  break;
58  }
59  return tmp.total_consumed();
60  }

203  void operator()(asio::error_code ec,
204  std::size_t bytes_transferred, int start = 0)
205  {
206  std::size_t max_size;
207  switch (start_ = start)
208  {
209  case 1:
210  max_size = this->check_for_completion(ec, buffers_.total_consumed());
211  for (;;)
212  {
213  {
214  ASIO_HANDLER_LOCATION((__FILE__, __LINE__, "async_write_at"));
215  device_.async_write_some_at(
216  offset_ + buffers_.total_consumed(), buffers_.prepare(max_size),
217  static_cast<write_at_op&&>(*this));
218  }
219  return; default:
220  buffers_.consume(bytes_transferred);
221  if ((!ec && bytes_transferred == 0) || buffers_.empty())
222  break;
223  max_size = this->check_for_completion(ec, buffers_.total_consumed());
224  if (max_size == 0)
225  break;
226  if (this->cancelled() != cancellation_type::none)
227  {
228  ec = asio::error::operation_aborted;
229  break;
230  }
231  }
232 
233  static_cast<WriteHandler&&>(handler_)(
234  static_cast<const asio::error_code&>(ec),
235  static_cast<const std::size_t&>(buffers_.total_consumed()));
236  }
237  }
