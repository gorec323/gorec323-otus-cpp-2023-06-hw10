35 // Wraps a handler to create an OVERLAPPED object for use with overlapped I/O.
36 class win_iocp_overlapped_ptr
37  : private noncopyable
38 {
39 public:
40  // Construct an empty win_iocp_overlapped_ptr.
41  win_iocp_overlapped_ptr()
42  : ptr_(0),
43  iocp_service_(0)
44  {
45  }
46 
47  // Construct an win_iocp_overlapped_ptr to contain the specified handler.
48  template <typename Executor, typename Handler>
49  explicit win_iocp_overlapped_ptr(const Executor& ex,
50  Handler&& handler)
51  : ptr_(0),
52  iocp_service_(0)
53  {
54  this->reset(ex, static_cast<Handler&&>(handler));
55  }
56 
57  // Destructor automatically frees the OVERLAPPED object unless released.
58  ~win_iocp_overlapped_ptr()
59  {
60  reset();
61  }
62 
63  // Reset to empty.
64  void reset()
65  {
66  if (ptr_)
67  {
68  ptr_->destroy();
69  ptr_ = 0;
70  iocp_service_->work_finished();
71  iocp_service_ = 0;
72  }
73  }
74 
75  // Reset to contain the specified handler, freeing any current OVERLAPPED
76  // object.
77  template <typename Executor, typename Handler>
78  void reset(const Executor& ex, Handler handler)
79  {
80  win_iocp_io_context* iocp_service = this->get_iocp_service(ex);
81 
82  typedef win_iocp_overlapped_op<Handler, Executor> op;
83  typename op::ptr p = { asio::detail::addressof(handler),
84  op::ptr::allocate(handler), 0 };
85  p.p = new (p.v) op(handler, ex);
86 
87  ASIO_HANDLER_CREATION((ex.context(), *p.p,
88  "iocp_service", iocp_service, 0, "overlapped"));
89 
90  iocp_service->work_started();
91  reset();
92  ptr_ = p.p;
93  p.v = p.p = 0;
94  iocp_service_ = iocp_service;
95  }
96 
97  // Get the contained OVERLAPPED object.
98  OVERLAPPED* get()
99  {
100  return ptr_;
101  }
102 
103  // Get the contained OVERLAPPED object.
104  const OVERLAPPED* get() const
105  {
106  return ptr_;
107  }
108 
109  // Release ownership of the OVERLAPPED object.
110  OVERLAPPED* release()
111  {
112  if (ptr_)
113  iocp_service_->on_pending(ptr_);
114 
115  OVERLAPPED* tmp = ptr_;
116  ptr_ = 0;
117  iocp_service_ = 0;
118  return tmp;
119  }
120 
121  // Post completion notification for overlapped operation. Releases ownership.
122  void complete(const asio::error_code& ec,
123  std::size_t bytes_transferred)
124  {
125  if (ptr_)
126  {
127  iocp_service_->on_completion(ptr_, ec,
128  static_cast<DWORD>(bytes_transferred));
129  ptr_ = 0;
130  iocp_service_ = 0;
131  }
132  }
133 
134 private:
135  template <typename Executor>
136  static win_iocp_io_context* get_iocp_service(const Executor& ex,
137  enable_if_t<
138  can_query<const Executor&, execution::context_t>::value
139  >* = 0)
140  {
141  return &use_service<win_iocp_io_context>(
142  asio::query(ex, execution::context));
143  }
144 
145  template <typename Executor>
146  static win_iocp_io_context* get_iocp_service(const Executor& ex,
147  enable_if_t<
148  !can_query<const Executor&, execution::context_t>::value
149  >* = 0)
150  {
151  return &use_service<win_iocp_io_context>(ex.context());
152  }
153 
154  static win_iocp_io_context* get_iocp_service(
155  const io_context::executor_type& ex)
156  {
157  return &asio::query(ex, execution::context).impl_;
158  }
159 
160  win_iocp_operation* ptr_;
161  win_iocp_io_context* iocp_service_;
162 };
