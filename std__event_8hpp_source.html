44  // Signal the event. (Retained for backward compatibility.)
45  template <typename Lock>
46  void signal(Lock& lock)
47  {
48  this->signal_all(lock);
49  }
50 
51  // Signal all waiters.
52  template <typename Lock>
53  void signal_all(Lock& lock)
54  {
55  ASIO_ASSERT(lock.locked());
56  (void)lock;
57  state_ |= 1;
58  cond_.notify_all();
59  }
60 
61  // Unlock the mutex and signal one waiter.
62  template <typename Lock>
63  void unlock_and_signal_one(Lock& lock)
64  {
65  ASIO_ASSERT(lock.locked());
66  state_ |= 1;
67  bool have_waiters = (state_ > 1);
68  lock.unlock();
69  if (have_waiters)
70  cond_.notify_one();
71  }
72 
73  // Unlock the mutex and signal one waiter who may destroy us.
74  template <typename Lock>
75  void unlock_and_signal_one_for_destruction(Lock& lock)
76  {
77  ASIO_ASSERT(lock.locked());
78  state_ |= 1;
79  bool have_waiters = (state_ > 1);
80  if (have_waiters)
81  cond_.notify_one();
82  lock.unlock();
83  }
84 
85  // If there's a waiter, unlock the mutex and signal it.
86  template <typename Lock>
87  bool maybe_unlock_and_signal_one(Lock& lock)
88  {
89  ASIO_ASSERT(lock.locked());
90  state_ |= 1;
91  if (state_ > 1)
92  {
93  lock.unlock();
94  cond_.notify_one();
95  return true;
96  }
97  return false;
98  }

100  // Reset the event.

102  void clear(Lock& lock)

104  ASIO_ASSERT(lock.locked());

109  // Wait for the event to become signalled.
110  template <typename Lock>
111  void wait(Lock& lock)
112  {
113  ASIO_ASSERT(lock.locked());
114  unique_lock_adapter u_lock(lock);
115  while ((state_ & 1) == 0)
116  {
117  waiter w(state_);
118  cond_.wait(u_lock.unique_lock_);
119  }
120  }
121 
122  // Timed wait for the event to become signalled.
123  template <typename Lock>
124  bool wait_for_usec(Lock& lock, long usec)
125  {
126  ASIO_ASSERT(lock.locked());
127  unique_lock_adapter u_lock(lock);
128  if ((state_ & 1) == 0)
129  {
130  waiter w(state_);
131  cond_.wait_for(u_lock.unique_lock_, std::chrono::microseconds(usec));
132  }
133  return (state_ & 1) != 0;
134  }
