217 template <typename Executor, typename Function, typename Allocator>
218 void strand_executor_service::do_execute(const implementation_type& impl,
219  Executor& ex, Function&& function, const Allocator& a)
220 {
221  typedef decay_t<Function> function_type;
222 
223  // If the executor is not never-blocking, and we are already in the strand,
224  // then the function can run immediately.
225  if (asio::query(ex, execution::blocking) != execution::blocking.never
226  && running_in_this_thread(impl))
227  {
228  // Make a local, non-const copy of the function.
229  function_type tmp(static_cast<Function&&>(function));
230 
231  fenced_block b(fenced_block::full);
232  static_cast<function_type&&>(tmp)();
233  return;
234  }
235 
236  // Allocate and construct an operation to wrap the function.
237  typedef executor_op<function_type, Allocator> op;
238  typename op::ptr p = { detail::addressof(a), op::ptr::allocate(a), 0 };
239  p.p = new (p.v) op(static_cast<Function&&>(function), a);
240 
241  ASIO_HANDLER_CREATION((impl->service_->context(), *p.p,
242  "strand_executor", impl.get(), 0, "execute"));
243 
244  // Add the function to the strand and schedule the strand if required.
245  bool first = enqueue(impl, p.p);
246  p.v = p.p = 0;
247  if (first)
248  {
249  ex.execute(invoker<Executor>(impl, ex));
250  }
251 }
