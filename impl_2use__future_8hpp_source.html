107 class promise_executor

109 public:

111  : p_(p)

116  {
117  return asio::query(system_executor(), execution::context);
118  }

121  {
122  return Blocking();
123  }
124 
125  promise_executor<T, execution::blocking_t::possibly_t>
126  require(execution::blocking_t::possibly_t) const
127  {

190 // The base class for all completion handlers that create promises.
191 template <typename T>
192 class promise_creator
193 {
194 public:
195  typedef promise_executor<T> executor_type;
196 
197  executor_type get_executor() const noexcept
198  {
199  return executor_type(p_);
200  }
201 
202  typedef std::future<T> future_type;
203 
204  future_type get_future()
205  {
206  return p_->get_future();
207  }
208 
209 protected:
210  template <typename Allocator>
211  void create_promise(const Allocator& a)
212  {
213  ASIO_REBIND_ALLOC(Allocator, char) b(a);
214  p_ = std::allocate_shared<std::promise<T>>(b, std::allocator_arg, b);
215  }
216 
217  shared_ptr<std::promise<T>> p_;
218 };
