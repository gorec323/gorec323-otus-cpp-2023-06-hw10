52  static status do_perform(reactor_op* base)
53  {
54  ASIO_ASSUME(base != 0);
55  reactive_socket_accept_op_base* o(
56  static_cast<reactive_socket_accept_op_base*>(base));
57 
58  socket_type new_socket = invalid_socket;
59  status result = socket_ops::non_blocking_accept(o->socket_,
60  o->state_, o->peer_endpoint_ ? o->peer_endpoint_->data() : 0,
61  o->peer_endpoint_ ? &o->addrlen_ : 0, o->ec_, new_socket)
62  ? done : not_done;
63  o->new_socket_.reset(new_socket);
64 
65  ASIO_HANDLER_REACTOR_OPERATION((*o, "non_blocking_accept", o->ec_));
66 
67  return result;
68  }
69 
70  void do_assign()
71  {
72  if (new_socket_.get() != invalid_socket)
73  {
74  if (peer_endpoint_)
75  peer_endpoint_->resize(addrlen_);
76  peer_.assign(protocol_, new_socket_.get(), ec_);
77  if (!ec_)
78  new_socket_.release();
79  }
80  }
81 
82 private:
83  socket_type socket_;
84  socket_ops::state_type state_;
85  socket_holder new_socket_;
86  Socket& peer_;
87  Protocol protocol_;
88  typename Protocol::endpoint* peer_endpoint_;
89  std::size_t addrlen_;
90 };

115  static void do_complete(void* owner, operation* base,
116  const asio::error_code& /*ec*/,
117  std::size_t /*bytes_transferred*/)
118  {
119  // Take ownership of the handler object.
120  ASIO_ASSUME(base != 0);
121  reactive_socket_accept_op* o(static_cast<reactive_socket_accept_op*>(base));
122  ptr p = { asio::detail::addressof(o->handler_), o, o };
123 
124  // On success, assign new connection to peer socket object.
125  if (owner)
126  o->do_assign();
127 
128  ASIO_HANDLER_COMPLETION((*o));
129 
130  // Take ownership of the operation's outstanding work.
131  handler_work<Handler, IoExecutor> w(
132  static_cast<handler_work<Handler, IoExecutor>&&>(
133  o->work_));
134 
135  ASIO_ERROR_LOCATION(o->ec_);
136 
137  // Make a copy of the handler so that the memory can be deallocated before
138  // the upcall is made. Even if we're not about to make an upcall, a
139  // sub-object of the handler may be the true owner of the memory associated
140  // with the handler. Consequently, a local copy of the handler is required
141  // to ensure that any owning sub-object remains valid until after we have
142  // deallocated the memory here.
143  detail::binder1<Handler, asio::error_code>
144  handler(o->handler_, o->ec_);
145  p.h = asio::detail::addressof(handler.handler_);
146  p.reset();
147 
148  // Make the upcall if required.
149  if (owner)
150  {
151  fenced_block b(fenced_block::half);
152  ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
153  w.complete(handler, handler.handler_);
154  ASIO_HANDLER_INVOCATION_END;
155  }
156  }
157

164 
165  // On success, assign new connection to peer socket object.
166  o->do_assign();
167 
168  ASIO_HANDLER_COMPLETION((*o));
169 
170  // Take ownership of the operation's outstanding work.
171  immediate_handler_work<Handler, IoExecutor> w(
172  static_cast<handler_work<Handler, IoExecutor>&&>(
173  o->work_));
174 
175  ASIO_ERROR_LOCATION(o->ec_);
176 
177  // Make a copy of the handler so that the memory can be deallocated before
178  // the upcall is made. Even if we're not about to make an upcall, a
179  // sub-object of the handler may be the true owner of the memory associated
180  // with the handler. Consequently, a local copy of the handler is required
181  // to ensure that any owning sub-object remains valid until after we have
182  // deallocated the memory here.
183  detail::binder1<Handler, asio::error_code>
184  handler(o->handler_, o->ec_);
185  p.h = asio::detail::addressof(handler.handler_);
186  p.reset();
187

191  }
