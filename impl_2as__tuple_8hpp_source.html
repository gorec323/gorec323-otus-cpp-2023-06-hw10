30 // Class to adapt a as_tuple_t as a completion handler.
31 template <typename Handler>
32 class as_tuple_handler
33 {
34 public:
35  typedef void result_type;
36 
37  template <typename CompletionToken>
38  as_tuple_handler(as_tuple_t<CompletionToken> e)
39  : handler_(static_cast<CompletionToken&&>(e.token_))
40  {
41  }
42 
43  template <typename RedirectedHandler>
44  as_tuple_handler(RedirectedHandler&& h)
45  : handler_(static_cast<RedirectedHandler&&>(h))
46  {
47  }
48 
49  template <typename... Args>
50  void operator()(Args&&... args)
51  {
52  static_cast<Handler&&>(handler_)(
53  std::make_tuple(static_cast<Args&&>(args)...));
54  }
55 
56 //private:
57  Handler handler_;
58 };
