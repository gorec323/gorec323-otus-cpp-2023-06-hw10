85  template<typename... T>
86  void operator()(T&&... t)
87  {
88  if (invocations_ < ~0u)
89  ++invocations_;
90  this->get_cancellation_state().slot().clear();
91  impl_(*this, static_cast<T&&>(t)...);
92  }
