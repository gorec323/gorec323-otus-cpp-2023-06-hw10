34 // Default polymorphic executor implementation.
35 template <typename Executor, typename Allocator>
36 class executor::impl
37  : public executor::impl_base
38 {
39 public:
40  typedef ASIO_REBIND_ALLOC(Allocator, impl) allocator_type;
41 
42  static impl_base* create(const Executor& e, Allocator a = Allocator())
43  {
44  raw_mem mem(a);
45  impl* p = new (mem.ptr_) impl(e, a);
46  mem.ptr_ = 0;
47  return p;
48  }
49 
50  impl(const Executor& e, const Allocator& a) noexcept
51  : impl_base(false),
52  ref_count_(1),
53  executor_(e),
54  allocator_(a)
55  {
56  }
57 
58  impl_base* clone() const noexcept
59  {
60  detail::ref_count_up(ref_count_);
61  return const_cast<impl_base*>(static_cast<const impl_base*>(this));
62  }
63 
64  void destroy() noexcept
65  {
66  if (detail::ref_count_down(ref_count_))
67  {
68  allocator_type alloc(allocator_);
69  impl* p = this;
70  p->~impl();
71  alloc.deallocate(p, 1);
72  }
73  }
74 
75  void on_work_started() noexcept
76  {
77  executor_.on_work_started();
78  }
79 
80  void on_work_finished() noexcept
81  {
82  executor_.on_work_finished();
83  }
84 
85  execution_context& context() noexcept
86  {
87  return executor_.context();
88  }
89 
90  void dispatch(function&& f)
91  {
92  executor_.dispatch(static_cast<function&&>(f), allocator_);
93  }
94 
95  void post(function&& f)
96  {
97  executor_.post(static_cast<function&&>(f), allocator_);
98  }
99 
100  void defer(function&& f)
101  {
102  executor_.defer(static_cast<function&&>(f), allocator_);
103  }
104 
105  type_id_result_type target_type() const noexcept
106  {
107  return type_id<Executor>();
108  }
109 
110  void* target() noexcept
111  {
112  return &executor_;
113  }
114 
115  const void* target() const noexcept
116  {
117  return &executor_;
118  }
119 
120  bool equals(const impl_base* e) const noexcept
121  {
122  if (this == e)
123  return true;
124  if (target_type() != e->target_type())
125  return false;
126  return executor_ == *static_cast<const Executor*>(e->target());
127  }
128 
129 private:
130  mutable detail::atomic_count ref_count_;
131  Executor executor_;
132  Allocator allocator_;
133 
134  struct raw_mem
135  {
136  allocator_type allocator_;
137  impl* ptr_;
138 
139  explicit raw_mem(const Allocator& a)
140  : allocator_(a),
141  ptr_(allocator_.allocate(1))
142  {
143  }
144 
145  ~raw_mem()
146  {
147  if (ptr_)
148  allocator_.deallocate(ptr_, 1);
149  }
150 
151  private:
152  // Disallow copying and assignment.
153  raw_mem(const raw_mem&);
154  raw_mem operator=(const raw_mem&);
155  };
