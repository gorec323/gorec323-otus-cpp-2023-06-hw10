27 // Lightweight, move-only function object wrapper.
28 class executor_function
29 {
30 public:
31  template <typename F, typename Alloc>
32  explicit executor_function(F f, const Alloc& a)
33  {
34  // Allocate and construct an object to wrap the function.
35  typedef impl<F, Alloc> impl_type;
36  typename impl_type::ptr p = {
37  detail::addressof(a), impl_type::ptr::allocate(a), 0 };
38  impl_ = new (p.v) impl_type(static_cast<F&&>(f), a);
39  p.v = 0;
40  }
41 
42  executor_function(executor_function&& other) noexcept
43  : impl_(other.impl_)
44  {
45  other.impl_ = 0;
46  }
47 
48  ~executor_function()
49  {
50  if (impl_)
51  impl_->complete_(impl_, false);
52  }
53 
54  void operator()()
55  {
56  if (impl_)
57  {
58  impl_base* i = impl_;
59  impl_ = 0;
60  i->complete_(i, true);
61  }
62  }
63 
64 private:
65  // Base class for polymorphic function implementations.
66  struct impl_base
67  {
68  void (*complete_)(impl_base*, bool);
69  };
70 
71  // Polymorphic function implementation.
72  template <typename Function, typename Alloc>
73  struct impl : impl_base
74  {
75  ASIO_DEFINE_TAGGED_HANDLER_ALLOCATOR_PTR(
76  thread_info_base::executor_function_tag, impl);
77 
78  template <typename F>
79  impl(F&& f, const Alloc& a)
80  : function_(static_cast<F&&>(f)),
81  allocator_(a)
82  {
83  complete_ = &executor_function::complete<Function, Alloc>;
84  }
85 
86  Function function_;
87  Alloc allocator_;
88  };
89 
90  // Helper to complete function invocation.
91  template <typename Function, typename Alloc>
92  static void complete(impl_base* base, bool call)
93  {
94  // Take ownership of the function object.
95  impl<Function, Alloc>* i(static_cast<impl<Function, Alloc>*>(base));
96  Alloc allocator(i->allocator_);
97  typename impl<Function, Alloc>::ptr p = {
98  detail::addressof(allocator), i, i };
99 
100  // Make a copy of the function so that the memory can be deallocated before
101  // the upcall is made. Even if we're not about to make an upcall, a
102  // sub-object of the function may be the true owner of the memory
103  // associated with the function. Consequently, a local copy of the function
104  // is required to ensure that any owning sub-object remains valid until
105  // after we have deallocated the memory here.
106  Function function(static_cast<Function&&>(i->function_));
107  p.reset();
108 
109  // Make the upcall if required.
110  if (call)
111  {
112  static_cast<Function&&>(function)();
113  }
114  }
115 
116  impl_base* impl_;
117 };
