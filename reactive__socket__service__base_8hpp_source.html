172  // Wait for the socket to become ready to read, ready to write, or to have
173  // pending error conditions.
174  asio::error_code wait(base_implementation_type& impl,
175  socket_base::wait_type w, asio::error_code& ec)
176  {
177  switch (w)
178  {
179  case socket_base::wait_read:
180  socket_ops::poll_read(impl.socket_, impl.state_, -1, ec);
181  break;
182  case socket_base::wait_write:
183  socket_ops::poll_write(impl.socket_, impl.state_, -1, ec);
184  break;
185  case socket_base::wait_error:
186  socket_ops::poll_error(impl.socket_, impl.state_, -1, ec);
187  break;
188  default:
189  ec = asio::error::invalid_argument;
190  break;
191  }
192 
193  return ec;
194  }

196  // Asynchronously wait for the socket to become ready to read, ready to
197  // write, or to have pending error conditions.
198  template <typename Handler, typename IoExecutor>
199  void async_wait(base_implementation_type& impl,
200  socket_base::wait_type w, Handler& handler, const IoExecutor& io_ex)
201  {
202  bool is_continuation =
203  asio_handler_cont_helpers::is_continuation(handler);
204 
205  associated_cancellation_slot_t<Handler> slot
206  = asio::get_associated_cancellation_slot(handler);
207 
208  // Allocate and construct an operation to wrap the handler.
209  typedef reactive_wait_op<Handler, IoExecutor> op;
210  typename op::ptr p = { asio::detail::addressof(handler),
211  op::ptr::allocate(handler), 0 };
212  p.p = new (p.v) op(success_ec_, handler, io_ex);
213 
214  ASIO_HANDLER_CREATION((reactor_.context(), *p.p, "socket",
215  &impl, impl.socket_, "async_wait"));
216 
217  int op_type;
218  switch (w)
219  {
220  case socket_base::wait_read:
221  op_type = reactor::read_op;
222  break;
223  case socket_base::wait_write:
224  op_type = reactor::write_op;
225  break;
226  case socket_base::wait_error:
227  op_type = reactor::except_op;
228  break;
229  default:
230  p.p->ec_ = asio::error::invalid_argument;
231  start_op(impl, reactor::read_op, p.p,
232  is_continuation, false, true, &io_ex, 0);
233  p.v = p.p = 0;
234  return;
235  }
236 
237  // Optionally register for per-operation cancellation.
238  if (slot.is_connected())
239  {
240  p.p->cancellation_key_ =
241  &slot.template emplace<reactor_op_cancellation>(
242  &reactor_, &impl.reactor_data_, impl.socket_, op_type);
243  }
244 
245  start_op(impl, op_type, p.p, is_continuation, false, false, &io_ex, 0);
246  p.v = p.p = 0;
247  }

249  // Send the given data to the peer.
250  template <typename ConstBufferSequence>
251  size_t send(base_implementation_type& impl,
252  const ConstBufferSequence& buffers,
253  socket_base::message_flags flags, asio::error_code& ec)
254  {
255  typedef buffer_sequence_adapter<asio::const_buffer,
256  ConstBufferSequence> bufs_type;
257 
258  if (bufs_type::is_single_buffer)
259  {
260  return socket_ops::sync_send1(impl.socket_,
261  impl.state_, bufs_type::first(buffers).data(),
262  bufs_type::first(buffers).size(), flags, ec);
263  }
264  else
265  {
266  bufs_type bufs(buffers);
267  return socket_ops::sync_send(impl.socket_, impl.state_,
268  bufs.buffers(), bufs.count(), flags, bufs.all_empty(), ec);
269  }
270  }
