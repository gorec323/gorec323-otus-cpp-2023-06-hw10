72 template <typename Payload, typename Handler, typename IoExecutor>
73 class channel_send_op : public channel_send<Payload>
74 {
75 public:
76  ASIO_DEFINE_HANDLER_PTR(channel_send_op);
77 
78  channel_send_op(Payload&& payload,
79  Handler& handler, const IoExecutor& io_ex)
80  : channel_send<Payload>(&channel_send_op::do_action,
81  static_cast<Payload&&>(payload)),
82  handler_(static_cast<Handler&&>(handler)),
83  work_(handler_, io_ex)
84  {
85  }
86 
87  static void do_action(channel_operation* base,
88  channel_operation::action a, void*)
89  {
90  // Take ownership of the operation object.
91  channel_send_op* o(static_cast<channel_send_op*>(base));
92  ptr p = { asio::detail::addressof(o->handler_), o, o };
93 
94  ASIO_HANDLER_COMPLETION((*o));
95 
96  // Take ownership of the operation's outstanding work.
97  channel_operation::handler_work<Handler, IoExecutor> w(
98  static_cast<channel_operation::handler_work<Handler, IoExecutor>&&>(
99  o->work_));
100 
101  asio::error_code ec;
102  switch (a)
103  {
104  case channel_operation::cancel_op:
105  ec = error::channel_cancelled;
106  break;
107  case channel_operation::close_op:
108  ec = error::channel_closed;
109  break;
110  default:
111  break;
112  }
113 
114  // Make a copy of the handler so that the memory can be deallocated before
115  // the handler is posted. Even if we're not about to post the handler, a
116  // sub-object of the handler may be the true owner of the memory associated
117  // with the handler. Consequently, a local copy of the handler is required
118  // to ensure that any owning sub-object remains valid until after we have
119  // deallocated the memory here.
120  asio::detail::binder1<Handler, asio::error_code>
121  handler(o->handler_, ec);
122  p.h = asio::detail::addressof(handler.handler_);
123  p.reset();
124 
125  // Post the completion if required.
126  if (a != channel_operation::destroy_op)
127  {
128  ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
129  if (a == channel_operation::immediate_op)
130  w.immediate(handler, handler.handler_, 0);
131  else
132  w.post(handler, handler.handler_);
133  ASIO_HANDLER_INVOCATION_END;
134  }
135  }
136 
137 private:
138  Handler handler_;
139  channel_operation::handler_work<Handler, IoExecutor> work_;
140 };
