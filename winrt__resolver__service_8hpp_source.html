113  // Resolve a query to a list of entries.
114  results_type resolve(implementation_type&,
115  const query_type& query, asio::error_code& ec)
116  {
117  try
118  {
119  using namespace Windows::Networking::Sockets;
120  auto endpoint_pairs = async_manager_.sync(
121  DatagramSocket::GetEndpointPairsAsync(
122  winrt_utils::host_name(query.host_name()),
123  winrt_utils::string(query.service_name())), ec);
124 
125  if (ec)
126  return results_type();
127 
128  return results_type::create(
129  endpoint_pairs, query.hints(),
130  query.host_name(), query.service_name());
131  }
132  catch (Platform::Exception^ e)
133  {
134  ec = asio::error_code(e->HResult,
135  asio::system_category());
136  return results_type();
137  }
138  }

140  // Asynchronously resolve a query to a list of entries.
141  template <typename Handler, typename IoExecutor>
142  void async_resolve(implementation_type& impl, const query_type& query,
143  Handler& handler, const IoExecutor& io_ex)
144  {
145  bool is_continuation =
146  asio_handler_cont_helpers::is_continuation(handler);
147 
148  // Allocate and construct an operation to wrap the handler.
149  typedef winrt_resolve_op<Protocol, Handler, IoExecutor> op;
150  typename op::ptr p = { asio::detail::addressof(handler),
151  op::ptr::allocate(handler), 0 };
152  p.p = new (p.v) op(query, handler, io_ex);
153 
154  ASIO_HANDLER_CREATION((scheduler_.context(),
155  *p.p, "resolver", &impl, 0, "async_resolve"));
156  (void)impl;
157 
158  try
159  {
160  using namespace Windows::Networking::Sockets;
161  async_manager_.async(DatagramSocket::GetEndpointPairsAsync(
162  winrt_utils::host_name(query.host_name()),
163  winrt_utils::string(query.service_name())), p.p);
164  p.v = p.p = 0;
165  }
166  catch (Platform::Exception^ e)
167  {
168  p.p->ec_ = asio::error_code(
169  e->HResult, asio::system_category());
170  scheduler_.post_immediate_completion(p.p, is_continuation);
171  p.v = p.p = 0;
172  }
173  }
174 
175  // Resolve an endpoint to a list of entries.
176  results_type resolve(implementation_type&,

178  {
179  ec = asio::error::operation_not_supported;
180  return results_type();
181  }

183  // Asynchronously resolve an endpoint to a list of entries.
184  template <typename Handler, typename IoExecutor>
185  void async_resolve(implementation_type&, const endpoint_type&,
186  Handler& handler, const IoExecutor& io_ex)
187  {
188  asio::error_code ec = asio::error::operation_not_supported;
189  const results_type results;
190  asio::post(io_ex, detail::bind_handler(handler, ec, results));
191  }
