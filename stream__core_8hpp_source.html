37  // sufficient to hold the largest possible TLS record.
38  enum { max_tls_record_size = 17 * 1024 };

40  template <typename Executor>
41  stream_core(SSL_CTX* context, const Executor& ex)
42  : engine_(context),
43  pending_read_(ex),
44  pending_write_(ex),
45  output_buffer_space_(max_tls_record_size),
46  output_buffer_(asio::buffer(output_buffer_space_)),
47  input_buffer_space_(max_tls_record_size),
48  input_buffer_(asio::buffer(input_buffer_space_))

104  stream_core& operator=(stream_core&& other)
105  {
106  if (this != &other)
107  {
108  engine_ = static_cast<engine&&>(other.engine_);
109 #if defined(ASIO_HAS_BOOST_DATE_TIME)

117  pending_read_ =

120  pending_write_ =
121  static_cast<asio::steady_timer&&>(
122  other.pending_write_);
123 #endif // defined(ASIO_HAS_BOOST_DATE_TIME)
124  output_buffer_space_ =
125  static_cast<std::vector<unsigned char>&&>(
126  other.output_buffer_space_);
127  output_buffer_ = other.output_buffer_;
128  input_buffer_space_ =
129  static_cast<std::vector<unsigned char>&&>(
130  other.input_buffer_space_);
131  input_buffer_ = other.input_buffer_;
132  input_ = other.input_;

134  other.input_buffer_ = asio::mutable_buffer(0, 0);
135  other.input_ = asio::const_buffer(0, 0);
136  }
137  return *this;
