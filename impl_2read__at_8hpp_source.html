40  template <typename SyncRandomAccessReadDevice, typename MutableBufferSequence,
41  typename MutableBufferIterator, typename CompletionCondition>
42  std::size_t read_at_buffer_sequence(SyncRandomAccessReadDevice& d,
43  uint64_t offset, const MutableBufferSequence& buffers,
44  const MutableBufferIterator&, CompletionCondition completion_condition,
45  asio::error_code& ec)
46  {
47  ec = asio::error_code();
48  asio::detail::consuming_buffers<mutable_buffer,
49  MutableBufferSequence, MutableBufferIterator> tmp(buffers);
50  while (!tmp.empty())
51  {
52  if (std::size_t max_size = detail::adapt_completion_condition_result(
53  completion_condition(ec, tmp.total_consumed())))
54  {
55  tmp.consume(d.read_some_at(offset + tmp.total_consumed(),
56  tmp.prepare(max_size), ec));
57  }
58  else
59  break;
60  }
61  return tmp.total_consumed();
62  }

111 template <typename SyncRandomAccessReadDevice, typename Allocator,
112  typename CompletionCondition>
113 std::size_t read_at(SyncRandomAccessReadDevice& d,
114  uint64_t offset, asio::basic_streambuf<Allocator>& b,
115  CompletionCondition completion_condition, asio::error_code& ec)
116 {
117  ec = asio::error_code();
118  std::size_t total_transferred = 0;
119  std::size_t max_size = detail::adapt_completion_condition_result(
120  completion_condition(ec, total_transferred));
121  std::size_t bytes_available = read_size_helper(b, max_size);
122  while (bytes_available > 0)
123  {
124  std::size_t bytes_transferred = d.read_some_at(
125  offset + total_transferred, b.prepare(bytes_available), ec);
126  b.commit(bytes_transferred);
127  total_transferred += bytes_transferred;
128  max_size = detail::adapt_completion_condition_result(
129  completion_condition(ec, total_transferred));
130  bytes_available = read_size_helper(b, max_size);
131  }
132  return total_transferred;
133 }

218  void operator()(asio::error_code ec,
219  std::size_t bytes_transferred, int start = 0)
220  {
221  std::size_t max_size;
222  switch (start_ = start)
223  {
224  case 1:
225  max_size = this->check_for_completion(ec, buffers_.total_consumed());
226  for (;;)
227  {
228  {
229  ASIO_HANDLER_LOCATION((__FILE__, __LINE__, "async_read_at"));
230  device_.async_read_some_at(
231  offset_ + buffers_.total_consumed(), buffers_.prepare(max_size),
232  static_cast<read_at_op&&>(*this));
233  }
234  return; default:
235  buffers_.consume(bytes_transferred);
236  if ((!ec && bytes_transferred == 0) || buffers_.empty())
237  break;
238  max_size = this->check_for_completion(ec, buffers_.total_consumed());
239  if (max_size == 0)
240  break;
241  if (this->cancelled() != cancellation_type::none)
242  {
243  ec = asio::error::operation_aborted;
244  break;
245  }
246  }
247 
248  static_cast<ReadHandler&&>(handler_)(
249  static_cast<const asio::error_code&>(ec),
250  static_cast<const std::size_t&>(buffers_.total_consumed()));
251  }
252  }
