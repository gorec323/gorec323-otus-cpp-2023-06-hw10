67  static void do_complete(void* owner, operation* base,
68  const asio::error_code& /*ec*/,
69  std::size_t /*bytes_transferred*/)
70  {
71  // Take ownership of the operation object.
72  ASIO_ASSUME(base != 0);
73  resolve_endpoint_op* o(static_cast<resolve_endpoint_op*>(base));
74  ptr p = { asio::detail::addressof(o->handler_), o, o };
75 
76  if (owner && owner != &o->scheduler_)
77  {
78  // The operation is being run on the worker io_context. Time to perform
79  // the resolver operation.
80 
81  // Perform the blocking endpoint resolution operation.
82  char host_name[NI_MAXHOST] = "";
83  char service_name[NI_MAXSERV] = "";
84  socket_ops::background_getnameinfo(o->cancel_token_, o->endpoint_.data(),
85  o->endpoint_.size(), host_name, NI_MAXHOST, service_name, NI_MAXSERV,
86  o->endpoint_.protocol().type(), o->ec_);
87  o->results_ = results_type::create(o->endpoint_, host_name, service_name);
88 
89  // Pass operation back to main io_context for completion.
90  o->scheduler_.post_deferred_completion(o);
91  p.v = p.p = 0;
92  }
93  else
94  {
95  // The operation has been returned to the main io_context. The completion
96  // handler is ready to be delivered.
97 
98  ASIO_HANDLER_COMPLETION((*o));
99 
100  // Take ownership of the operation's outstanding work.
101  handler_work<Handler, IoExecutor> w(
102  static_cast<handler_work<Handler, IoExecutor>&&>(
103  o->work_));
104 
105  // Make a copy of the handler so that the memory can be deallocated
106  // before the upcall is made. Even if we're not about to make an upcall,
107  // a sub-object of the handler may be the true owner of the memory
108  // associated with the handler. Consequently, a local copy of the handler
109  // is required to ensure that any owning sub-object remains valid until
110  // after we have deallocated the memory here.
111  detail::binder2<Handler, asio::error_code, results_type>
112  handler(o->handler_, o->ec_, o->results_);
113  p.h = asio::detail::addressof(handler.handler_);
114  p.reset();
115 
116  if (owner)
117  {
118  fenced_block b(fenced_block::half);
119  ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, "..."));
120  w.complete(handler, handler.handler_);
121  ASIO_HANDLER_INVOCATION_END;
122  }
123  }
124  }
125

130  Handler handler_;

132  results_type results_;
133 };
