159  struct initiate_async_wait
160  {
161  std::shared_ptr<detail::promise_impl<
162  void(Ts...), Executor, Allocator>> self_;
163 
164  template <typename WaitHandler>
165  void operator()(WaitHandler&& handler) const
166  {
167  const auto alloc = get_associated_allocator(
168  handler, self_->get_allocator());
169 
170  auto cancel = get_associated_cancellation_slot(handler);
171 
172  if (self_->done)
173  {
174  auto exec = asio::get_associated_executor(
175  handler, self_->get_executor());
176 
177  asio::post(exec,
178  [self = std::move(self_),
179  handler = std::forward<WaitHandler>(handler)]() mutable
180  {
181  self->apply(std::move(handler));
182  });
183  }
184  else

195  void(Ts...), Executor, Allocator>> self)
196  : self(std::move(self))
197  {
198  }

201  {

204  p->cancel.emit(level);
