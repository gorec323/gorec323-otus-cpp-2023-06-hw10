40  template <typename SyncReadStream, typename MutableBufferSequence,
41  typename MutableBufferIterator, typename CompletionCondition>
42  std::size_t read_buffer_seq(SyncReadStream& s,
43  const MutableBufferSequence& buffers, const MutableBufferIterator&,
44  CompletionCondition completion_condition, asio::error_code& ec)
45  {
46  ec = asio::error_code();
47  asio::detail::consuming_buffers<mutable_buffer,
48  MutableBufferSequence, MutableBufferIterator> tmp(buffers);
49  while (!tmp.empty())
50  {
51  if (std::size_t max_size = detail::adapt_completion_condition_result(
52  completion_condition(ec, tmp.total_consumed())))
53  tmp.consume(s.read_some(tmp.prepare(max_size), ec));
54  else
55  break;
56  }
57  return tmp.total_consumed();
58  }

244  typename CompletionCondition>
245 std::size_t read(SyncReadStream& s, DynamicBuffer_v2 buffers,
246  CompletionCondition completion_condition, asio::error_code& ec,
247  constraint_t<
248  is_dynamic_buffer_v2<DynamicBuffer_v2>::value
249  >)
250 {
251  DynamicBuffer_v2& b = buffers;
252 
253  ec = asio::error_code();
254  std::size_t total_transferred = 0;
255  std::size_t max_size = detail::adapt_completion_condition_result(
256  completion_condition(ec, total_transferred));
257  std::size_t bytes_available = std::min<std::size_t>(
258  std::max<std::size_t>(512, b.capacity() - b.size()),
259  std::min<std::size_t>(max_size, b.max_size() - b.size()));
260  while (bytes_available > 0)
261  {

263  b.grow(bytes_available);
264  std::size_t bytes_transferred = s.read_some(
265  b.data(pos, bytes_available), ec);
266  b.shrink(bytes_available - bytes_transferred);
267  total_transferred += bytes_transferred;
268  max_size = detail::adapt_completion_condition_result(
269  completion_condition(ec, total_transferred));
270  bytes_available = std::min<std::size_t>(
271  std::max<std::size_t>(512, b.capacity() - b.size()),
272  std::min<std::size_t>(max_size, b.max_size() - b.size()));
273  }
274  return total_transferred;

362  std::size_t bytes_transferred, int start = 0)
363  {
364  std::size_t max_size;
365  switch (start_ = start)
366  {
367  case 1:
368  max_size = this->check_for_completion(ec, buffers_.total_consumed());
369  for (;;)
370  {
371  {
372  ASIO_HANDLER_LOCATION((__FILE__, __LINE__, "async_read"));
373  stream_.async_read_some(buffers_.prepare(max_size),
374  static_cast<read_op&&>(*this));
375  }

379  break;

381  if (max_size == 0)
382  break;
383  if (this->cancelled() != cancellation_type::none)
384  {
385  ec = error::operation_aborted;
386  break;
387  }
388  }

391  static_cast<const asio::error_code&>(ec),
392  static_cast<const std::size_t&>(buffers_.total_consumed()));
393  }
