39 public:

46  : has_value_(other.has_value_)

48  if (has_value_)
49  new (&u_.value_) T(std::move(other.get()));
50  }
51

53  {
54  if (has_value_)
55  u_.value_.~T();
56  }
57 
58  T& get() noexcept
59  {
60  return u_.value_;
61  }
62 
63  template <typename... Args>

68  }

71  union u
72  {
73  u() {}
74  ~u() {}
75  char c_;
76  T value_;
77  } u_;
78  bool has_value_;

85 struct parallel_group_completion_handler
86 {

88  prefer_result_t<
89  associated_executor_t<Handler>,
90  execution::outstanding_work_t::tracked_t
91  >
92  > executor_type;
93 
94  parallel_group_completion_handler(Handler&& h)
95  : handler_(std::move(h)),
96  executor_(
97  asio::prefer(
98  asio::get_associated_executor(handler_),
99  execution::outstanding_work.tracked))
100  {
101  }
102 
103  executor_type get_executor() const noexcept
104  {
105  return executor_;
106  }
107 
108  void operator()()
109  {
110  this->invoke(asio::detail::make_index_sequence<sizeof...(Ops)>());
111  }
112 
113  template <std::size_t... I>
114  void invoke(asio::detail::index_sequence<I...>)

117  }

145 // Shared state for the parallel group.
146 template <typename Condition, typename Handler, typename... Ops>
147 struct parallel_group_state
148 {
149  parallel_group_state(Condition&& c, Handler&& h)
150  : cancellation_condition_(std::move(c)),
151  handler_(std::move(h))
152  {
153  }
154 
155  // The number of operations that have completed so far. Used to determine the
156  // order of completion.
157  std::atomic<unsigned int> completed_{0};
158 
159  // The non-none cancellation type that resulted from a cancellation condition.
160  // Stored here for use by the group's initiating function.
161  std::atomic<cancellation_type_t> cancel_type_{cancellation_type::none};
162 
163  // The number of cancellations that have been requested, either on completion
164  // of the operations within the group, or via the cancellation slot for the
165  // group operation. Initially set to the number of operations to prevent
166  // cancellation signals from being emitted until after all of the group's
167  // operations' initiating functions have completed.
168  std::atomic<unsigned int> cancellations_requested_{sizeof...(Ops)};
169 
170  // The number of operations that are yet to complete. Used to determine when
171  // it is safe to invoke the user's completion handler.
172  std::atomic<unsigned int> outstanding_{sizeof...(Ops)};
173 
174  // The cancellation signals for each operation in the group.
175  asio::cancellation_signal cancellation_signals_[sizeof...(Ops)];
176 
177  // The cancellation condition is used to determine whether the results from an
178  // individual operation warrant a cancellation request for the whole group.
179  Condition cancellation_condition_;
180 
181  // The proxy handler to be invoked once all operations in the group complete.
182  parallel_group_completion_handler<Handler, Ops...> handler_;
183 };

185 // Handler for an individual operation within the parallel group.
186 template <std::size_t I, typename Condition, typename Handler, typename... Ops>
187 struct parallel_group_op_handler
188 {
189  typedef asio::cancellation_slot cancellation_slot_type;
190 
191  parallel_group_op_handler(
192  std::shared_ptr<parallel_group_state<Condition, Handler, Ops...>> state)
193  : state_(std::move(state))
194  {
195  }
196 
197  cancellation_slot_type get_cancellation_slot() const noexcept
198  {
199  return state_->cancellation_signals_[I].slot();
200  }
201 
202  template <typename... Args>
203  void operator()(Args... args)
204  {
205  // Capture this operation into the completion order.
206  state_->handler_.completion_order_[state_->completed_++] = I;
207 
208  // Determine whether the results of this operation require cancellation of
209  // the whole group.
210  cancellation_type_t cancel_type = state_->cancellation_condition_(args...);
211 
212  // Capture the result of the operation into the proxy completion handler.
213  std::get<I>(state_->handler_.args_).emplace(std::move(args)...);
214 
215  if (cancel_type != cancellation_type::none)
216  {
217  // Save the type for potential use by the group's initiating function.
218  state_->cancel_type_ = cancel_type;
219 
220  // If we are the first operation to request cancellation, emit a signal
221  // for each operation in the group.
222  if (state_->cancellations_requested_++ == 0)
223  for (std::size_t i = 0; i < sizeof...(Ops); ++i)
224  if (i != I)
225  state_->cancellation_signals_[i].emit(cancel_type);
226  }
227 
228  // If this is the last outstanding operation, invoke the user's handler.
229  if (--state_->outstanding_ == 0)
230  asio::dispatch(std::move(state_->handler_));
231  }
232 
233  std::shared_ptr<parallel_group_state<Condition, Handler, Ops...>> state_;
234 };

359 template <typename Condition, typename Handler,
360  typename... Ops, std::size_t... I>
361 void parallel_group_launch(Condition cancellation_condition, Handler handler,
362  std::tuple<Ops...>& ops, asio::detail::index_sequence<I...>)
363 {
364  // Get the user's completion handler's cancellation slot, so that we can allow
365  // cancellation of the entire group.
366  associated_cancellation_slot_t<Handler> slot
367  = asio::get_associated_cancellation_slot(handler);
368 
369  // Create the shared state for the operation.
370  typedef parallel_group_state<Condition, Handler, Ops...> state_type;
371  std::shared_ptr<state_type> state = std::allocate_shared<state_type>(
372  asio::detail::recycling_allocator<state_type,
373  asio::detail::thread_info_base::parallel_group_tag>(),
374  std::move(cancellation_condition), std::move(handler));
375 
376  // Initiate each individual operation in the group.
377  int fold[] = { 0,
378  ( parallel_group_op_launcher<I, Ops>::launch(std::get<I>(ops), state),
379  0 )...
380  };
381  (void)fold;
382 
383  // Check if any of the operations has already requested cancellation, and if
384  // so, emit a signal for each operation in the group.
385  if ((state->cancellations_requested_ -= sizeof...(Ops)) > 0)
386  for (auto& signal : state->cancellation_signals_)
387  signal.emit(state->cancel_type_);
388 
389  // Register a handler with the user's completion handler's cancellation slot.
390  if (slot.is_connected())
391  slot.template emplace<
392  parallel_group_cancellation_handler<
393  Condition, Handler, Ops...>>(state);
394 }

396 // Proxy completion handler for the ranged group of parallel operatations.
397 // Unpacks and recombines the individual operations' results, and invokes the
398 // user's completion handler.
399 template <typename Handler, typename Op, typename Allocator>
400 struct ranged_parallel_group_completion_handler
401 {
402  typedef decay_t<
403  prefer_result_t<
404  associated_executor_t<Handler>,
405  execution::outstanding_work_t::tracked_t
406  >
407  > executor_type;
408 
409  typedef typename parallel_op_signature_as_tuple<
410  completion_signature_of_t<Op>
411  >::type op_tuple_type;
412 
413  typedef parallel_group_op_result<op_tuple_type> op_result_type;
414 
415  ranged_parallel_group_completion_handler(Handler&& h,
416  std::size_t size, const Allocator& allocator)
417  : handler_(std::move(h)),
418  executor_(
419  asio::prefer(
420  asio::get_associated_executor(handler_),
421  execution::outstanding_work.tracked)),
422  allocator_(allocator),
423  completion_order_(size, 0,
424  ASIO_REBIND_ALLOC(Allocator, std::size_t)(allocator)),
425  args_(ASIO_REBIND_ALLOC(Allocator, op_result_type)(allocator))
426  {
427  for (std::size_t i = 0; i < size; ++i)
428  args_.emplace_back();
429  }
430 
431  executor_type get_executor() const noexcept
432  {
433  return executor_;
434  }
435 
436  void operator()()
437  {
438  this->invoke(
439  asio::detail::make_index_sequence<
440  std::tuple_size<op_tuple_type>::value>());
441  }
442 
443  template <std::size_t... I>
444  void invoke(asio::detail::index_sequence<I...>)
445  {
446  typedef typename parallel_op_signature_as_tuple<
447  typename ranged_parallel_group_signature<
448  completion_signature_of_t<Op>,
449  Allocator
450  >::raw_type
451  >::type vectors_type;
452 
453  // Construct all result vectors using the supplied allocator.
454  vectors_type vectors{
455  typename std::tuple_element<I, vectors_type>::type(
456  ASIO_REBIND_ALLOC(Allocator, int)(allocator_))...};
457 
458  // Reserve sufficient space in each of the result vectors.
459  int reserve_fold[] = { 0,
460  ( std::get<I>(vectors).reserve(completion_order_.size()),
461  0 )...
462  };
463  (void)reserve_fold;
464 
465  // Copy the results from all operations into the result vectors.
466  for (std::size_t idx = 0; idx < completion_order_.size(); ++idx)
467  {
468  int pushback_fold[] = { 0,
469  ( std::get<I>(vectors).push_back(
470  std::move(std::get<I>(args_[idx].get()))),
471  0 )...
472  };
473  (void)pushback_fold;
474  }
475 
476  std::move(handler_)(completion_order_, std::move(std::get<I>(vectors))...);
477  }
478 
479  Handler handler_;
480  executor_type executor_;
481  Allocator allocator_;
482  std::vector<std::size_t,
483  ASIO_REBIND_ALLOC(Allocator, std::size_t)> completion_order_;
484  std::deque<op_result_type,
485  ASIO_REBIND_ALLOC(Allocator, op_result_type)> args_;
486 };

685 template <typename Condition, typename Handler,
686  typename Range, typename Allocator>
687 void ranged_parallel_group_launch(Condition cancellation_condition,
688  Handler handler, Range&& range, const Allocator& allocator)
689 {
690  // Get the user's completion handler's cancellation slot, so that we can allow
691  // cancellation of the entire group.
692  associated_cancellation_slot_t<Handler> slot
693  = asio::get_associated_cancellation_slot(handler);
694 
695  // The type of the asynchronous operation.
696  typedef decay_t<decltype(*declval<typename Range::iterator>())> op_type;
697 
698  // Create the shared state for the operation.
699  typedef ranged_parallel_group_state<Condition,
700  Handler, op_type, Allocator> state_type;
701  std::shared_ptr<state_type> state = std::allocate_shared<state_type>(
702  asio::detail::recycling_allocator<state_type,
703  asio::detail::thread_info_base::parallel_group_tag>(),
704  std::move(cancellation_condition),
705  std::move(handler), range.size(), allocator);
706 
707  std::size_t idx = 0;
708  for (auto&& op : std::forward<Range>(range))
709  {
710  typedef associated_executor_t<op_type> ex_type;
711  ex_type ex = asio::get_associated_executor(op);
712  std::move(op)(
713  ranged_parallel_group_op_handler_with_executor<
714  ex_type, Condition, Handler, op_type, Allocator>(
715  state, std::move(ex), idx++));
716  }
717 
718  // Check if any of the operations has already requested cancellation, and if
719  // so, emit a signal for each operation in the group.
720  if ((state->cancellations_requested_ -= range.size()) > 0)
721  for (auto& signal : state->cancellation_signals_)
722  signal.emit(state->cancel_type_);
723 
724  // Register a handler with the user's completion handler's cancellation slot.
725  if (slot.is_connected())
726  slot.template emplace<
727  ranged_parallel_group_cancellation_handler<
728  Condition, Handler, op_type, Allocator>>(state);
729 }
