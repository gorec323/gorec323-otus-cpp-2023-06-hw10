142  template <typename CancellationHandler, typename... Args>
143  CancellationHandler& emplace(Args&&... args)
144  {
145  typedef detail::cancellation_handler<CancellationHandler>
146  cancellation_handler_type;
147  auto_delete_helper del = { prepare_memory(
148  sizeof(cancellation_handler_type),
149  alignof(CancellationHandler)) };
150  cancellation_handler_type* handler_obj =
151  new (del.mem.first) cancellation_handler_type(
152  del.mem.second, static_cast<Args&&>(args)...);
153  del.mem.first = 0;
154  *handler_ = handler_obj;
155  return handler_obj->handler();
156  }
