102 // Helper class to translate buffers into the native buffer representation.
103 template <typename Buffer, typename Buffers>
104 class buffer_sequence_adapter
105  : buffer_sequence_adapter_base
106 {
107 public:
108  enum { is_single_buffer = false };
109  enum { is_registered_buffer = false };
110 
111  explicit buffer_sequence_adapter(const Buffers& buffer_sequence)
112  : count_(0), total_buffer_size_(0)
113  {
114  buffer_sequence_adapter::init(
115  asio::buffer_sequence_begin(buffer_sequence),
116  asio::buffer_sequence_end(buffer_sequence));
117  }
118 
119  native_buffer_type* buffers()
120  {
121  return buffers_;
122  }
123 
124  std::size_t count() const
125  {
126  return count_;
127  }
128 
129  std::size_t total_size() const
130  {
131  return total_buffer_size_;
132  }
133 
134  registered_buffer_id registered_id() const
135  {
136  return registered_buffer_id();
137  }
138 
139  bool all_empty() const
140  {
141  return total_buffer_size_ == 0;
142  }
143 
144  static bool all_empty(const Buffers& buffer_sequence)
145  {
146  return buffer_sequence_adapter::all_empty(
147  asio::buffer_sequence_begin(buffer_sequence),
148  asio::buffer_sequence_end(buffer_sequence));
149  }
150 
151  static void validate(const Buffers& buffer_sequence)
152  {
153  buffer_sequence_adapter::validate(
154  asio::buffer_sequence_begin(buffer_sequence),
155  asio::buffer_sequence_end(buffer_sequence));
156  }
157 
158  static Buffer first(const Buffers& buffer_sequence)
159  {
160  return buffer_sequence_adapter::first(
161  asio::buffer_sequence_begin(buffer_sequence),
162  asio::buffer_sequence_end(buffer_sequence));
163  }
164 
165  enum { linearisation_storage_size = 8192 };
166 
167  static Buffer linearise(const Buffers& buffer_sequence,
168  const asio::mutable_buffer& storage)
169  {
170  return buffer_sequence_adapter::linearise(
171  asio::buffer_sequence_begin(buffer_sequence),
172  asio::buffer_sequence_end(buffer_sequence), storage);
173  }
174 
175 private:
176  template <typename Iterator>
177  void init(Iterator begin, Iterator end)
178  {
179  Iterator iter = begin;
180  for (; iter != end && count_ < max_buffers; ++iter, ++count_)
181  {
182  Buffer buffer(*iter);
183  init_native_buffer(buffers_[count_], buffer);
184  total_buffer_size_ += buffer.size();
185  }
186  }
187 
188  template <typename Iterator>
189  static bool all_empty(Iterator begin, Iterator end)
190  {
191  Iterator iter = begin;
192  std::size_t i = 0;
193  for (; iter != end && i < max_buffers; ++iter, ++i)
194  if (Buffer(*iter).size() > 0)
195  return false;
196  return true;
197  }
198 
199  template <typename Iterator>
200  static void validate(Iterator begin, Iterator end)
201  {
202  Iterator iter = begin;
203  for (; iter != end; ++iter)
204  {
205  Buffer buffer(*iter);
206  buffer.data();
207  }
208  }
209 
210  template <typename Iterator>
211  static Buffer first(Iterator begin, Iterator end)
212  {
213  Iterator iter = begin;
214  for (; iter != end; ++iter)
215  {
216  Buffer buffer(*iter);
217  if (buffer.size() != 0)
218  return buffer;
219  }
220  return Buffer();
221  }
222 
223  template <typename Iterator>
224  static Buffer linearise(Iterator begin, Iterator end,
225  const asio::mutable_buffer& storage)
226  {
227  asio::mutable_buffer unused_storage = storage;
228  Iterator iter = begin;
229  while (iter != end && unused_storage.size() != 0)
230  {
231  Buffer buffer(*iter);
232  ++iter;
233  if (buffer.size() == 0)
234  continue;
235  if (unused_storage.size() == storage.size())
236  {
237  if (iter == end)
238  return buffer;
239  if (buffer.size() >= unused_storage.size())
240  return buffer;
241  }
242  unused_storage += asio::buffer_copy(unused_storage, buffer);
243  }
244  return Buffer(storage.data(), storage.size() - unused_storage.size());
245  }
246 
247  native_buffer_type buffers_[max_buffers];
248  std::size_t count_;
249  std::size_t total_buffer_size_;
250 };
