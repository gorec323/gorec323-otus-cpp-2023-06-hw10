667  template <typename Executor>
668  void execute_and_wait(Executor&& ex)
669  {
670  handler h = { this };
671  ex.execute(h);
672  asio::detail::mutex::scoped_lock lock(mutex_);
673  while (!is_complete_)
674  event_.wait(lock);
675  }
676 
677  struct cleanup
678  {
679  ~cleanup()
680  {
681  asio::detail::mutex::scoped_lock lock(state_->mutex_);
682  state_->is_complete_ = true;
683  state_->event_.unlock_and_signal_one_for_destruction(lock);
684  }
685 
686  blocking_execute_state* state_;
687  };
688 
689  struct handler
690  {
691  void operator()()
692  {
693  cleanup c = { state_ };
694  state_->func_();
695  }
696 
697  blocking_execute_state* state_;
698  };

700  Function func_;
701  asio::detail::mutex mutex_;
702  asio::detail::event event_;
703  bool is_complete_;
