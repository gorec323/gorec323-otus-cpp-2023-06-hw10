61  static void do_prepare(io_uring_operation* base, ::io_uring_sqe* sqe)
62  {
63  ASIO_ASSUME(base != 0);
64  io_uring_socket_recvfrom_op_base* o(
65  static_cast<io_uring_socket_recvfrom_op_base*>(base));
66 
67  if ((o->state_ & socket_ops::internal_non_blocking) != 0)
68  {
69  bool except_op = (o->flags_ & socket_base::message_out_of_band) != 0;
70  ::io_uring_prep_poll_add(sqe, o->socket_, except_op ? POLLPRI : POLLIN);
71  }
72  else
73  {
74  ::io_uring_prep_recvmsg(sqe, o->socket_, &o->msghdr_, o->flags_);
75  }
76  }

78  static bool do_perform(io_uring_operation* base, bool after_completion)
79  {
80  ASIO_ASSUME(base != 0);
81  io_uring_socket_recvfrom_op_base* o(
82  static_cast<io_uring_socket_recvfrom_op_base*>(base));
83 
84  if ((o->state_ & socket_ops::internal_non_blocking) != 0)
85  {
86  bool except_op = (o->flags_ & socket_base::message_out_of_band) != 0;
87  if (after_completion || !except_op)
88  {
89  std::size_t addr_len = o->sender_endpoint_.capacity();
90  bool result;
91  if (o->bufs_.is_single_buffer)
92  {
93  result = socket_ops::non_blocking_recvfrom1(o->socket_,
94  o->bufs_.first(o->buffers_).data(),
95  o->bufs_.first(o->buffers_).size(), o->flags_,
96  o->sender_endpoint_.data(), &addr_len,
97  o->ec_, o->bytes_transferred_);
98  }
99  else
100  {
101  result = socket_ops::non_blocking_recvfrom(o->socket_,
102  o->bufs_.buffers(), o->bufs_.count(), o->flags_,
103  o->sender_endpoint_.data(), &addr_len,
104  o->ec_, o->bytes_transferred_);
105  }
106  if (result && !o->ec_)
107  o->sender_endpoint_.resize(addr_len);
108  }
109  }
110  else if (after_completion && !o->ec_)
111  o->sender_endpoint_.resize(o->msghdr_.msg_namelen);
112 
113  if (o->ec_ && o->ec_ == asio::error::would_block)
114  {
115  o->state_ |= socket_ops::internal_non_blocking;
116  return false;
117  }
118 
119  return after_completion;
120  }
121 
122 private:
123  socket_type socket_;
124  socket_ops::state_type state_;
125  MutableBufferSequence buffers_;
126  Endpoint& sender_endpoint_;
127  socket_base::message_flags flags_;
128  buffer_sequence_adapter<asio::mutable_buffer,
129  MutableBufferSequence> bufs_;
130  msghdr msghdr_;
131 };

154  static void do_complete(void* owner, operation* base,
155  const asio::error_code& /*ec*/,
156  std::size_t /*bytes_transferred*/)
157  {
158  // Take ownership of the handler object.
159  ASIO_ASSUME(base != 0);
160  io_uring_socket_recvfrom_op* o
161  (static_cast<io_uring_socket_recvfrom_op*>(base));
162  ptr p = { asio::detail::addressof(o->handler_), o, o };
163 
164  ASIO_HANDLER_COMPLETION((*o));
165 
166  // Take ownership of the operation's outstanding work.
167  handler_work<Handler, IoExecutor> w(
168  static_cast<handler_work<Handler, IoExecutor>&&>(
169  o->work_));
170 
171  ASIO_ERROR_LOCATION(o->ec_);
172 
173  // Make a copy of the handler so that the memory can be deallocated before
174  // the upcall is made. Even if we're not about to make an upcall, a
175  // sub-object of the handler may be the true owner of the memory associated
176  // with the handler. Consequently, a local copy of the handler is required
177  // to ensure that any owning sub-object remains valid until after we have
178  // deallocated the memory here.
179  detail::binder2<Handler, asio::error_code, std::size_t>
180  handler(o->handler_, o->ec_, o->bytes_transferred_);
181  p.h = asio::detail::addressof(handler.handler_);
182  p.reset();
183 
184  // Make the upcall if required.
185  if (owner)
186  {
187  fenced_block b(fenced_block::half);
188  ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, handler.arg2_));
189  w.complete(handler, handler.handler_);
190  ASIO_HANDLER_INVOCATION_END;
191  }
192  }
193 
194 private:
195  Handler handler_;
196  handler_work<Handler, IoExecutor> work_;
197 };
