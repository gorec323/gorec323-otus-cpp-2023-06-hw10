222 template <typename Protocol, typename Executor,
223  typename Iterator, typename ConnectCondition>
224 Iterator connect(basic_socket<Protocol, Executor>& s, Iterator begin,
225  Iterator end, ConnectCondition connect_condition,
226  asio::error_code& ec)
227 {
228  ec = asio::error_code();
229 
230  for (Iterator iter = begin; iter != end; ++iter)
231  {
232  iter = (detail::call_connect_condition(connect_condition, ec, iter, end));
233  if (iter != end)
234  {
235  s.close(ec);
236  s.connect(*iter, ec);
237  if (!ec)
238  return iter;
239  }
240  else
241  break;
242  }
243 
244  if (!ec)
245  ec = asio::error::not_found;
246 
247  return end;
248 }

343  //private:
344  template <typename Iterator>
345  void process(asio::error_code ec,
346  int start, Iterator begin, Iterator end)
347  {
348  Iterator iter = begin;
349  std::advance(iter, index_);
350 
351  switch (start_ = start)
352  {
353  case 1:
354  for (;;)
355  {
356  this->check_condition(ec, iter, end);
357  index_ = std::distance(begin, iter);
358 
359  if (iter != end)
360  {
361  socket_.close(ec);
362  ASIO_HANDLER_LOCATION((__FILE__, __LINE__, "async_connect"));
363  socket_.async_connect(*iter,
364  static_cast<range_connect_op&&>(*this));
365  return;
366  }
367 
368  if (start)
369  {
370  ec = asio::error::not_found;
371  ASIO_HANDLER_LOCATION((__FILE__, __LINE__, "async_connect"));
372  asio::post(socket_.get_executor(),
373  detail::bind_handler(
374  static_cast<range_connect_op&&>(*this), ec));
375  return;
376  }
377 
378  /* fall-through */ default:
379 
380  if (iter == end)
381  break;
382 
383  if (!socket_.is_open())
384  {
385  ec = asio::error::operation_aborted;
386  break;
387  }
388 
389  if (!ec)
390  break;
391 
392  if (this->cancelled() != cancellation_type::none)
393  {
394  ec = asio::error::operation_aborted;
395  break;
396  }
397 
398  ++iter;
399  ++index_;
400  }
401 
402  static_cast<RangeConnectHandler&&>(handler_)(
403  static_cast<const asio::error_code&>(ec),
404  static_cast<const typename Protocol::endpoint&>(
405  ec || iter == end ? typename Protocol::endpoint() : *iter));
406  }
407  }
408 
409  basic_socket<Protocol, Executor>& socket_;
410  EndpointSequence endpoints_;
411  std::size_t index_;
412  int start_;
413  RangeConnectHandler handler_;
414  };
