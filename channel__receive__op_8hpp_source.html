58 template <typename Payload, typename Handler, typename IoExecutor>
59 class channel_receive_op : public channel_receive<Payload>
60 {
61 public:
62  ASIO_DEFINE_HANDLER_PTR(channel_receive_op);
63 
64  template <typename... Args>
65  channel_receive_op(Handler& handler, const IoExecutor& io_ex)
66  : channel_receive<Payload>(&channel_receive_op::do_action),
67  handler_(static_cast<Handler&&>(handler)),
68  work_(handler_, io_ex)
69  {
70  }
71 
72  static void do_action(channel_operation* base,
73  channel_operation::action a, void* v)
74  {
75  // Take ownership of the operation object.
76  channel_receive_op* o(static_cast<channel_receive_op*>(base));
77  ptr p = { asio::detail::addressof(o->handler_), o, o };
78 
79  ASIO_HANDLER_COMPLETION((*o));
80 
81  // Take ownership of the operation's outstanding work.
82  channel_operation::handler_work<Handler, IoExecutor> w(
83  static_cast<channel_operation::handler_work<Handler, IoExecutor>&&>(
84  o->work_));
85 
86  // Make a copy of the handler so that the memory can be deallocated before
87  // the handler is posted. Even if we're not about to post the handler, a
88  // sub-object of the handler may be the true owner of the memory associated
89  // with the handler. Consequently, a local copy of the handler is required
90  // to ensure that any owning sub-object remains valid until after we have
91  // deallocated the memory here.
92  if (a != channel_operation::destroy_op)
93  {
94  Payload* payload = static_cast<Payload*>(v);
95  channel_handler<Payload, Handler> handler(
96  static_cast<Payload&&>(*payload), o->handler_);
97  p.h = asio::detail::addressof(handler.handler_);
98  p.reset();
99  ASIO_HANDLER_INVOCATION_BEGIN(());
100  if (a == channel_operation::immediate_op)
101  w.immediate(handler, handler.handler_, 0);
102  else if (a == channel_operation::dispatch_op)
103  w.dispatch(handler, handler.handler_);
104  else
105  w.post(handler, handler.handler_);
106  ASIO_HANDLER_INVOCATION_END;
107  }
108  else
109  {
110  asio::detail::binder0<Handler> handler(o->handler_);
111  p.h = asio::detail::addressof(handler.handler_);
112  p.reset();
113  }
114  }
115 
116 private:
117  Handler handler_;
118  channel_operation::handler_work<Handler, IoExecutor> work_;
119 };
