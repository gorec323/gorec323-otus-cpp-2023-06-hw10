42 thread_pool::basic_executor_type<Allocator, Bits>&
43 thread_pool::basic_executor_type<Allocator, Bits>::operator=(
44  const basic_executor_type& other) noexcept
45 {
46  if (this != &other)
47  {
48  thread_pool* old_thread_pool = pool_;
49  pool_ = other.pool_;
50  allocator_ = other.allocator_;
51  bits_ = other.bits_;
52  if (Bits & outstanding_work_tracked)
53  {
54  if (pool_)
55  pool_->scheduler_.work_started();

58  }
59  }
60  return *this;
61 }

91 template <typename Allocator, unsigned int Bits>
92 template <typename Function>
93 void thread_pool::basic_executor_type<Allocator,
94  Bits>::do_execute(Function&& f, false_type) const
95 {
96  typedef decay_t<Function> function_type;
97 
98  // Invoke immediately if the blocking.possibly property is enabled and we are
99  // already inside the thread pool.
100  if ((bits_ & blocking_never) == 0 && pool_->scheduler_.can_dispatch())
101  {
102  // Make a local, non-const copy of the function.
103  function_type tmp(static_cast<Function&&>(f));
104 
105 #if !defined(ASIO_NO_EXCEPTIONS)
106  try
107  {
108 #endif // !defined(ASIO_NO_EXCEPTIONS)
109  detail::fenced_block b(detail::fenced_block::full);
110  static_cast<function_type&&>(tmp)();
111  return;
112 #if !defined(ASIO_NO_EXCEPTIONS)
113  }
114  catch (...)
115  {
116  pool_->scheduler_.capture_current_exception();
117  return;
118  }
119 #endif // !defined(ASIO_NO_EXCEPTIONS)
120  }
121 
122  // Allocate and construct an operation to wrap the function.
123  typedef detail::executor_op<function_type, Allocator> op;
124  typename op::ptr p = { detail::addressof(allocator_),
125  op::ptr::allocate(allocator_), 0 };
126  p.p = new (p.v) op(static_cast<Function&&>(f), allocator_);
127 
128  if ((bits_ & relationship_continuation) != 0)
129  {
130  ASIO_HANDLER_CREATION((*pool_, *p.p,
131  "thread_pool", pool_, 0, "execute(blk=never,rel=cont)"));
132  }
133  else
134  {
135  ASIO_HANDLER_CREATION((*pool_, *p.p,
136  "thread_pool", pool_, 0, "execute(blk=never,rel=fork)"));
137  }
138 
139  pool_->scheduler_.post_immediate_completion(p.p,
140  (bits_ & relationship_continuation) != 0);
141  p.v = p.p = 0;
142 }

144 template <typename Allocator, unsigned int Bits>
145 template <typename Function>
146 void thread_pool::basic_executor_type<Allocator,
147  Bits>::do_execute(Function&& f, true_type) const
148 {
149  // Obtain a non-const instance of the function.
150  detail::non_const_lvalue<Function> f2(f);
151 
152  // Invoke immediately if we are already inside the thread pool.
153  if (pool_->scheduler_.can_dispatch())
154  {
155 #if !defined(ASIO_NO_EXCEPTIONS)
156  try
157  {
158 #endif // !defined(ASIO_NO_EXCEPTIONS)
159  detail::fenced_block b(detail::fenced_block::full);
160  static_cast<decay_t<Function>&&>(f2.value)();
161  return;
162 #if !defined(ASIO_NO_EXCEPTIONS)
163  }
164  catch (...)
165  {
166  std::terminate();
167  }
168 #endif // !defined(ASIO_NO_EXCEPTIONS)
169  }
170 
171  // Construct an operation to wrap the function.
172  typedef decay_t<Function> function_type;
173  detail::blocking_executor_op<function_type> op(f2.value);
174 
175  ASIO_HANDLER_CREATION((*pool_, op,
176  "thread_pool", pool_, 0, "execute(blk=always)"));
177 
178  pool_->scheduler_.post_immediate_completion(&op, false);
179  op.wait();
180 }
