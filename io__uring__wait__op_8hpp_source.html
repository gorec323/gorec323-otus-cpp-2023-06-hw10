61  static void do_complete(void* owner, operation* base,
62  const asio::error_code& /*ec*/,
63  std::size_t /*bytes_transferred*/)
64  {
65  // Take ownership of the handler object.
66  ASIO_ASSUME(base != 0);
67  io_uring_wait_op* o(static_cast<io_uring_wait_op*>(base));
68  ptr p = { asio::detail::addressof(o->handler_), o, o };
69 
70  ASIO_HANDLER_COMPLETION((*o));
71 
72  // Take ownership of the operation's outstanding work.
73  handler_work<Handler, IoExecutor> w(
74  static_cast<handler_work<Handler, IoExecutor>&&>(
75  o->work_));
76 
77  ASIO_ERROR_LOCATION(o->ec_);
78 
79  // Make a copy of the handler so that the memory can be deallocated before
80  // the upcall is made. Even if we're not about to make an upcall, a
81  // sub-object of the handler may be the true owner of the memory associated
82  // with the handler. Consequently, a local copy of the handler is required
83  // to ensure that any owning sub-object remains valid until after we have
84  // deallocated the memory here.
85  detail::binder1<Handler, asio::error_code>
86  handler(o->handler_, o->ec_);
87  p.h = asio::detail::addressof(handler.handler_);
88  p.reset();
89 
90  // Make the upcall if required.
91  if (owner)
92  {
93  fenced_block b(fenced_block::half);
94  ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, handler.arg2_));
95  w.complete(handler, handler.handler_);
96  ASIO_HANDLER_INVOCATION_END;
97  }
98  }
99 
100 private:
101  Handler handler_;
102  handler_work<Handler, IoExecutor> work_;
103  int descriptor_;
104  int poll_flags_;
105 };
