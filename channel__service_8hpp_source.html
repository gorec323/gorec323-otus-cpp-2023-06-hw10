492  Traits, R(asio::error_code)>
493  : channel_service::base_implementation_type

496  typedef typename Traits::template rebind<R(asio::error_code)>::other

498 
499  // Type of an element stored in the buffer.
500  typedef conditional_t<
501  has_signature<
502  typename traits_type::receive_cancelled_signature,
503  R(asio::error_code)
504  >::value,
505  conditional_t<
506  has_signature<
507  typename traits_type::receive_closed_signature,
508  R(asio::error_code)
509  >::value,
510  channel_payload<R(asio::error_code)>,
511  channel_payload<

514  >
515  >,
516  conditional_t<
517  has_signature<
518  typename traits_type::receive_closed_signature,
519  R(asio::error_code),
520  typename traits_type::receive_cancelled_signature
521  >::value,
522  channel_payload<
523  R(asio::error_code),
524  typename traits_type::receive_cancelled_signature
525  >,
526  channel_payload<
527  R(asio::error_code),
528  typename traits_type::receive_cancelled_signature,
529  typename traits_type::receive_closed_signature
530  >
531  >
532  > payload_type;
533 
534  // Construct with empty buffer.
535  implementation_type()
536  : size_(0)
537  {
538  first_.count_ = 0;
539  }
540 
541  // Move from another buffer.
542  void buffer_move_from(implementation_type& other)
543  {
544  size_ = other.buffer_;
545  other.size_ = 0;
546  first_ = other.first_;
547  other.first.count_ = 0;
548  rest_ = static_cast<
549  typename traits_type::template container<buffered_value>::type&&>(
550  other.rest_);
551  other.buffer_clear();
552  }
553 
554  // Get number of buffered elements.
555  std::size_t buffer_size() const
556  {
557  return size_;
558  }
559 
560  // Push a new value to the back of the buffer.
561  void buffer_push(payload_type payload)
562  {
563  buffered_value& last = rest_.empty() ? first_ : rest_.back();
564  if (last.count_ == 0)
565  {
566  value_handler handler{last.value_};
567  payload.receive(handler);
568  last.count_ = 1;
569  }
570  else
571  {
572  asio::error_code value{last.value_};
573  value_handler handler{value};
574  payload.receive(handler);
575  if (last.value_ == value)
576  ++last.count_;
577  else
578  rest_.push_back({value, 1});
579  }
580  ++size_;
581  }
582 
583  // Push new values to the back of the buffer.
584  std::size_t buffer_push_n(std::size_t count, payload_type payload)
585  {
586  std::size_t available = this->max_buffer_size_ - size_;
587  count = (count < available) ? count : available;
588  if (count > 0)
589  {
590  buffered_value& last = rest_.empty() ? first_ : rest_.back();
591  if (last.count_ == 0)
592  {
593  payload.receive(value_handler{last.value_});
594  last.count_ = count;
595  }
596  else
597  {
598  asio::error_code value{last.value_};
599  payload.receive(value_handler{value});
600  if (last.value_ == value)
601  last.count_ += count;
602  else
603  rest_.push_back({value, count});
604  }
605  size_ += count;
606  }
607  return count;
608  }
609 
610  // Get the element at the front of the buffer.
611  payload_type buffer_front()
612  {
613  return payload_type({0, first_.value_});
614  }
615 
616  // Pop a value from the front of the buffer.
617  void buffer_pop()
618  {
619  --size_;
620  if (--first_.count_ == 0 && !rest_.empty())
621  {
622  first_ = rest_.front();
623  rest_.pop_front();
624  }
625  }
626 
627  // Clear all values from the buffer.
628  void buffer_clear()
629  {
630  size_ = 0;
631  first_.count_ == 0;
632  rest_.clear();
633  }
634 
635 private:
636  struct buffered_value
637  {
638  asio::error_code value_;
639  std::size_t count_;
640  };
641 
642  struct value_handler
643  {
644  asio::error_code& target_;
645 
646  template <typename... Args>
647  void operator()(const asio::error_code& value, Args&&...)
648  {
649  target_ = value;
650  }
651  };
652 
653  buffered_value& last_value()
654  {
655  return rest_.empty() ? first_ : rest_.back();
656  }
657 
658  // Total number of buffered values.
659  std::size_t size_;
660 
661  // The first buffered value is maintained as a separate data member to avoid
662  // allocating space in the container in the common case.
663  buffered_value first_;
664 
665  // The rest of the buffered values.
666  typename traits_type::template container<buffered_value>::type rest_;
667 };
