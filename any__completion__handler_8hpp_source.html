131  void* allocate(std::size_t size, std::size_t align) const
132  {
133  typename std::allocator_traits<
134  associated_allocator_t<Handler,
135  asio::recycling_allocator<void>>>::template
136  rebind_alloc<unsigned char> alloc(
137  (get_associated_allocator)(handler_,
138  asio::recycling_allocator<void>()));
139 
140  std::size_t space = size + align - 1;
141  unsigned char* base =
142  std::allocator_traits<decltype(alloc)>::allocate(
143  alloc, space + sizeof(std::ptrdiff_t));
144 
145  void* p = base;
146  if (detail::align(align, size, p, space))
147  {
148  std::ptrdiff_t off = static_cast<unsigned char*>(p) - base;
149  std::memcpy(static_cast<unsigned char*>(p) + size, &off, sizeof(off));
150  return p;
151  }
152 
153  std::bad_alloc ex;
154  asio::detail::throw_exception(ex);
155  return nullptr;
156  }
157 
158  void deallocate(void* p, std::size_t size, std::size_t align) const
159  {
160  if (p)
161  {
162  typename std::allocator_traits<
163  associated_allocator_t<Handler,
164  asio::recycling_allocator<void>>>::template
165  rebind_alloc<unsigned char> alloc(
166  (get_associated_allocator)(handler_,
167  asio::recycling_allocator<void>()));
168 
169  std::ptrdiff_t off;
170  std::memcpy(&off, static_cast<unsigned char*>(p) + size, sizeof(off));
171  unsigned char* base = static_cast<unsigned char*>(p) - off;
172 
173  std::allocator_traits<decltype(alloc)>::deallocate(
174  alloc, base, size + align -1 + sizeof(std::ptrdiff_t));
175  }
176  }

178  template <typename... Args>
179  void call(Args&&... args)
180  {
181  deleter d{
182  (get_associated_allocator)(handler_,
183  asio::recycling_allocator<void>())};
184 
185  std::unique_ptr<any_completion_handler_impl, deleter> ptr(this, d);
186  Handler handler(static_cast<Handler&&>(handler_));
187  ptr.reset();
188 
189  static_cast<Handler&&>(handler)(
190  static_cast<Args&&>(args)...);
191  }
192 
193 private:
194  Handler handler_;
195 };
