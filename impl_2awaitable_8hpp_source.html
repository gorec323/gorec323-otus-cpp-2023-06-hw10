48 // An awaitable_thread represents a thread-of-execution that is composed of one
49 // or more "stack frames", with each frame represented by an awaitable_frame.
50 // All execution occurs in the context of the awaitable_thread's executor. An
51 // awaitable_thread continues to "pump" the stack frames by repeatedly resuming
52 // the top stack frame until the stack is empty, or until ownership of the
53 // stack is transferred to another awaitable_thread object.
54 //
55 // +------------------------------------+
56 // | top_of_stack_ |
57 // | V
58 // +--------------+---+ +-----------------+
59 // | | | |
60 // | awaitable_thread |<---------------------------+ awaitable_frame |
61 // | | attached_thread_ | |
62 // +--------------+---+ (Set only when +---+-------------+
63 // | frames are being |
64 // | actively pumped | caller_
65 // | by a thread, and |
66 // | then only for V
67 // | the top frame.) +-----------------+
68 // | | |
69 // | | awaitable_frame |
70 // | | |
71 // | +---+-------------+
72 // | |
73 // | | caller_
74 // | :
75 // | :
76 // | |
77 // | V
78 // | +-----------------+
79 // | bottom_of_stack_ | |
80 // +------------------------------->| awaitable_frame |
81 // | |
82 // +-----------------+

663  // Construct from the entry point of a new thread of execution.
664  awaitable_thread(awaitable<awaitable_thread_entry_point, Executor> p,

667  : bottom_of_stack_(std::move(p))
668  {
669  bottom_of_stack_.frame_->top_of_stack_ = bottom_of_stack_.frame_;
670  new (&bottom_of_stack_.frame_->u_.executor_) Executor(ex);
671  bottom_of_stack_.frame_->has_executor_ = true;
672  bottom_of_stack_.frame_->parent_cancellation_slot_ = parent_cancel_slot;

676  // Transfer ownership from another awaitable_thread.
677  awaitable_thread(awaitable_thread&& other) noexcept
678  : bottom_of_stack_(std::move(other.bottom_of_stack_))
679  {
680  }
681 
682  // Clean up with a last ditch effort to ensure the thread is unwound within
683  // the context of the executor.
684  ~awaitable_thread()
685  {
686  if (bottom_of_stack_.valid())
687  {
688  // Coroutine "stack unwinding" must be performed through the executor.
689  auto* bottom_frame = bottom_of_stack_.frame_;
690  (post)(bottom_frame->u_.executor_,
691  [a = std::move(bottom_of_stack_)]() mutable
692  {
693  (void)awaitable<awaitable_thread_entry_point, Executor>(
694  std::move(a));
695  });
696  }
697  }

706  return bottom_of_stack_.frame_->u_.executor_;

710  {

712  }

718  }

763  // Repeatedly resume the top stack frame until the stack is empty or until it
764  // has been transferred to another resumable_thread object.
765  void pump()
766  {
767  do
768  bottom_of_stack_.frame_->top_of_stack_->resume();
769  while (bottom_of_stack_.frame_ && bottom_of_stack_.frame_->top_of_stack_);
770 
771  if (bottom_of_stack_.frame_)
772  {
773  awaitable<awaitable_thread_entry_point, Executor> a(
774  std::move(bottom_of_stack_));
775  a.frame_->rethrow_exception();
776  }
777  }
