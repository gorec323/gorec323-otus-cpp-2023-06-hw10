76  static void do_complete(void* owner, operation* base,
77  const asio::error_code& result_ec,
78  std::size_t /*bytes_transferred*/)
79  {
80  asio::error_code ec(result_ec);
81 
82  // Take ownership of the operation object.
83  ASIO_ASSUME(base != 0);
84  win_iocp_socket_connect_op* o(
85  static_cast<win_iocp_socket_connect_op*>(base));
86  ptr p = { asio::detail::addressof(o->handler_), o, o };
87 
88  if (owner)
89  {
90  if (o->connect_ex_)
91  socket_ops::complete_iocp_connect(o->socket_, ec);
92  else
93  ec = o->ec_;
94  }
95 
96  ASIO_HANDLER_COMPLETION((*o));
97 
98  // Take ownership of the operation's outstanding work.
99  handler_work<Handler, IoExecutor> w(
100  static_cast<handler_work<Handler, IoExecutor>&&>(
101  o->work_));
102 
103  ASIO_ERROR_LOCATION(ec);
104 
105  // Make a copy of the handler so that the memory can be deallocated before
106  // the upcall is made. Even if we're not about to make an upcall, a
107  // sub-object of the handler may be the true owner of the memory associated
108  // with the handler. Consequently, a local copy of the handler is required
109  // to ensure that any owning sub-object remains valid until after we have
110  // deallocated the memory here.
111  detail::binder1<Handler, asio::error_code>
112  handler(o->handler_, ec);
113  p.h = asio::detail::addressof(handler.handler_);
114  p.reset();
115 
116  // Make the upcall if required.
117  if (owner)
118  {
119  fenced_block b(fenced_block::half);
120  ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
121  w.complete(handler, handler.handler_);
122  ASIO_HANDLER_INVOCATION_END;
123  }
124  }
125 
126 private:
127  Handler handler_;
128  handler_work<Handler, IoExecutor> work_;
129 };
