231  {
232  if (ec)

243  std::size_t length = bytes_avail < space_avail

245  storage_.resize(orig_size + length);
246  const std::size_t bytes_copied = asio::buffer_copy(
247  storage_.data() + orig_size, buffers_, length);
248  static_cast<WriteHandler&&>(handler_)(ec, bytes_copied);
