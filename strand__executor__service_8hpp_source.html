35 // Default service implementation for a strand.
36 class strand_executor_service
37  : public execution_context_service_base<strand_executor_service>

39 public:
40  // The underlying implementation of a strand.
41  class strand_impl
42  {
43  public:
44  ASIO_DECL ~strand_impl();
45 
46  private:
47  friend class strand_executor_service;
48 
49  // Mutex to protect access to internal data.
50  mutex* mutex_;
51 
52  // Indicates whether the strand is currently "locked" by a handler. This
53  // means that there is a handler upcall in progress, or that the strand
54  // itself has been scheduled in order to invoke some pending handlers.
55  bool locked_;
56 
57  // Indicates that the strand has been shut down and will accept no further
58  // handlers.
59  bool shutdown_;
60 
61  // The handlers that are waiting on the strand but should not be run until
62  // after the next time the strand is scheduled. This queue must only be
63  // modified while the mutex is locked.
64  op_queue<scheduler_operation> waiting_queue_;
65 
66  // The handlers that are ready to be run. Logically speaking, these are the
67  // handlers that hold the strand's lock. The ready queue is only modified
68  // from within the strand and so may be accessed without locking the mutex.
69  op_queue<scheduler_operation> ready_queue_;
70 
71  // Pointers to adjacent handle implementations in linked list.
72  strand_impl* next_;
73  strand_impl* prev_;
74 
75  // The strand service in where the implementation is held.
76  strand_executor_service* service_;
77  };
