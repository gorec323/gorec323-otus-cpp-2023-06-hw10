54 inline ASIO_SYNC_OP_VOID connect_pair(
55  basic_socket<Protocol, Executor1>& socket1,
56  basic_socket<Protocol, Executor2>& socket2, asio::error_code& ec)
57 {
58  // Check that this function is only being used with a UNIX domain socket.
59  asio::local::basic_endpoint<Protocol>* tmp
60  = static_cast<typename Protocol::endpoint*>(0);
61  (void)tmp;
62 
63  Protocol protocol;
64  asio::detail::socket_type sv[2];
65  if (asio::detail::socket_ops::socketpair(protocol.family(),
66  protocol.type(), protocol.protocol(), sv, ec)
67  == asio::detail::socket_error_retval)
68  ASIO_SYNC_OP_VOID_RETURN(ec);
69 
70  socket1.assign(protocol, sv[0], ec);
71  if (ec)
72  {
73  asio::error_code temp_ec;
74  asio::detail::socket_ops::state_type state[2] = { 0, 0 };
75  asio::detail::socket_ops::close(sv[0], state[0], true, temp_ec);
76  asio::detail::socket_ops::close(sv[1], state[1], true, temp_ec);
77  ASIO_SYNC_OP_VOID_RETURN(ec);
78  }
79 
80  socket2.assign(protocol, sv[1], ec);
81  if (ec)
82  {
83  asio::error_code temp_ec;
84  socket1.close(temp_ec);
85  asio::detail::socket_ops::state_type state = 0;
86  asio::detail::socket_ops::close(sv[1], state, true, temp_ec);
87  ASIO_SYNC_OP_VOID_RETURN(ec);
88  }
89 
90  ASIO_SYNC_OP_VOID_RETURN(ec);
91 }
